import { useState } from 'react';

function CodeSnippet({ code }) {
  const [copied, setCopied] = useState(false);

  const handleCopyClick = () => {
    navigator.clipboard.writeText(code);
    setCopied(true);
  };

  return (
    <div>
      <pre>{code}</pre>
      <button onClick={handleCopyClick}>{copied ? 'Copi√© !' : 'Copier'}</button>
    </div>
  );
}
